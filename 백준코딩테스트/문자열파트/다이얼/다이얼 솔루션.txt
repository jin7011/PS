#include <stdio.h>
#include <string>

int Switch(char s) {

	int num = 0;

	switch (s) {
	case 'A':
	case 'B':
	case 'C':
		num = 2;
		break;
	case 'D':
	case 'E':
	case 'F':
		num = 3;
		break;
	case 'G':
	case 'H':
	case 'I':
		num = 4;
		break;
	case 'J':
	case 'K':
	case 'L':
		num = 5;
		break;
	case 'M':
	case 'N':
	case 'O':
		num = 6;
		break;
	case 'P':
	case 'Q':
	case 'R':
	case 'S':
		num = 7;
		break;
	case 'T':
	case 'U':
	case 'V':
		num = 8;
		break;
	case 'W':
	case 'X':
	case 'Y':
	case 'Z':
		num = 9;
		break;

	default:
		break;
	}
	return num+1;
}

int main() {

	char S[1000001] = { 0 };
	int total = 0;

	scanf_s("%s", S, sizeof(S));
	int len = strlen(S);

	for (int x = 0; x < len; x++)
		total = total + Switch(S[x]);

	printf("%d", total);


	return 0;
}
